is_independent_china_or_warlord = { #Does not include sinkiang
	OR = {
		tag = CHI
		tag = PRC
		tag = GXC
		tag = YUN
		tag = SHX
		tag = XSM
	}
	is_puppet = no
}

is_alone_or_in_faction_at_peace_with_CHI = {
	OR = {
		is_in_faction = no
		is_in_faction_with = CHI
		AND = {
			is_in_faction = yes
			NOT = {
				is_in_faction_with = CHI
			}
		}
	}
	NOT = {
		any_country = {
			is_in_faction_with = PREV
			has_war_with = CHI
		}
	}
}

has_war_with_internal = {
	NOT = {
		has_war_with = PRC
		has_war_with = SHX
		has_war_with = GXC
		has_war_with = XSM
		has_war_with = YUN
	}
}

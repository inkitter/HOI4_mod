    trigger = {
       any_owned_state = {
           memorialplaza > 0
       }
    }

    mean_time_to_happen = { days = 0 }

    immediate = {
        hidden_effect = { 
            remove_ideas = eqexp_plazatocore
        }
    }

    option = {
        name = techexp.33.a
        ai_chance = { factor = 1 }
        random_list = {
            70 = {
                random_state = {
                    limit = {
                        AND= {
                            is_owned_by = ROOT
                            memorialplaza > 0
                            NOT = { is_core_of = ROOT }
                        }
                    }
                    remove_building = {
                        type = memorialplaza 
                        level = 1
                    }
                    add_core_of = ROOT
                }
            }

                 random_state = {
                    limit = {
                        AND= {
                            is_owned_by = ROOT

                            NOT = { is_claimed_by = ROOT }
                        }
                    }
                    remove_building = {
                        type = memorialplaza 
                        level = 1
                    }
                    add_claim_by = ROOT
                }
            }
        }
    }
    option = {
        name = techexp.33.b
